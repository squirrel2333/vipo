```java
package com.hqw.mybatis.mapper;

import com.hqw.mybatis.pojo.User;
import org.apache.ibatis.annotations.MapKey;
import org.apache.ibatis.annotations.Param;

import java.util.List;
import java.util.Map;

public interface SelectMapper {

    List<User> getUserById(@Param("id") Integer id);

    List<User> getAllUser();

    Integer getCount();

    Map<String, Object> getUserByIdToMap(@Param("id") Integer id);

    List<Map<String, Object>> getAllUserToMap();

    @MapKey("id")
    Map<String, Object> getAllUserToMapByMapKey();
}
```

```java
package com.hqw.mybais.mapper;

import com.hqw.mybatis.mapper.SelectMapper;
import com.hqw.mybatis.pojo.User;
import com.hqw.mybatis.utils.SqlSessionUtils;
import org.apache.ibatis.session.SqlSession;
import org.junit.Test;

import java.util.List;
import java.util.Map;

public class SelectMapperTest {

    /**
     * 1. 若查询的数据有一条，可以通过实体类对象 或者 集合接收
     * User{id=1, username='hqw', password='123', age=12, sex=b, email='sss@sss'}
     * [User{id=1, username='hqw', password='123', age=12, sex=b, email='sss@sss'}]
     */
    @Test
    public void testGetUserById() {
        SqlSession sqlSession = SqlSessionUtils.getSqlSession();
        SelectMapper mapper = sqlSession.getMapper(SelectMapper.class);
        List<User> user = mapper.getUserById(1);
        System.out.println(user);
    }

    /**
     * 2. 若查询的数据有多条，不能通过实体类对象来接收，此时会抛异常TooManyResultsException
     *    可以以实体类类型的List集合接收
     *    [User{id=1, username='hqw', password='123', age=12, sex=b, email='sss@sss'}, User{id=2, username='www', password='123', age=12, sex=b, email='sss@sss'}]
     */
    @Test
    public void testGetAllUser() {
        SqlSession sqlSession = SqlSessionUtils.getSqlSession();
        SelectMapper mapper = sqlSession.getMapper(SelectMapper.class);
        List<User> user = mapper.getAllUser();
        System.out.println(user);
    }

    /**
     * 3. 查询的结果为其它类型的，mybatis为resultType设置了类型别名
     */
    @Test
    public void testCount() {
        SqlSession sqlSession = SqlSessionUtils.getSqlSession();
        SelectMapper mapper = sqlSession.getMapper(SelectMapper.class);
        Integer num = mapper.getCount();
        System.out.println(num);
    }

    /**
     * 4. 若查询的数据有一条，可以通过Map接收
     *    结果会以查询的字段为键，字段对应的值为值
     *    {password=123, sex=b, id=2, age=12, email=sss@sss, username=www}
     */
    @Test
    public void testGetUserByIdToMap() {
        SqlSession sqlSession = SqlSessionUtils.getSqlSession();
        SelectMapper mapper = sqlSession.getMapper(SelectMapper.class);
        Map<String, Object> user = mapper.getUserByIdToMap(2);
        System.out.println(user);
    }

    /**
     * 5. 若查询的数据有多条，不能通过实体类对象的Map来接收，此时会抛异常TooManyResultsException
     *    可以以Map类型的List集合接收
     *    [{password=123, sex=b, id=1, age=12, email=sss@sss, username=hqw}, {password=123, sex=b, id=2, age=12, email=sss@sss, username=www}]
     */
    @Test
    public void testGetAllUserToMap() {
        SqlSession sqlSession = SqlSessionUtils.getSqlSession();
        SelectMapper mapper = sqlSession.getMapper(SelectMapper.class);
        List<Map<String, Object>> users = mapper.getAllUserToMap();
        System.out.println(users);
    }

    /**
     * 6. 若查询的数据有多条
     *    可以在mapper接口的方法上添加注解@MapKey(“key")
     *    结果每条数据会以字段key对应的值为 map的键，其余字段对应的值为map的值，组成一个map集合、
     *    {1={password=123, sex=b, id=1, age=12, email=sss@sss, username=hqw}, 2={password=123, sex=b, id=2, age=12, email=sss@sss, username=www}}
     */
    @Test
    public void getAllUserToMapByMapKey() {
        SqlSession sqlSession = SqlSessionUtils.getSqlSession();
        SelectMapper mapper = sqlSession.getMapper(SelectMapper.class);
        Map<String, Object> users = mapper.getAllUserToMapByMapKey();
        System.out.println(users);
    }
}
```
