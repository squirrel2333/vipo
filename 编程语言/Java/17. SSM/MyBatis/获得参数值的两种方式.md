```java
<?xml version="1.0" encoding="UTF-8" ?>
<!DOCTYPE mapper
        PUBLIC "-//mybatis.org//DTD Mapper 3.0//EN"
        "http://mybatis.org/dtd/mybatis-3-mapper.dtd">

<mapper namespace="com.hqw.mybatis.mapper.ParameterMapper">

    <!-- List<User> getAllUser();-->
    <select id="getAllUser" resultType="User">
        select * from t_user
    </select>

    <!--User getUserByName(String name);-->
    <select id="getUserByName" resultType="User">
        <!--#{} 和${} 中的名称username与参数名称无关，可以写成任何名称。-->
        <!-- select * from t_user where username = '${username}'-->
        select * from t_user where username = #{username}
    </select>

    <!--User checkLogin(String username, String password);-->
    <select id="checkLogin" resultType="User">
        select * from t_user where username = '${param1}' and password = #{arg1}
        <!-- select * from t_user where username = #{arg0} and password = #{arg1}-->
    </select>

    <!--User checkLoginByMap(Map map);-->
    <select id="checkLoginByMap" resultType="User">
        select * from t_user where username = #{username} and password=#{password}
    </select>

    <!--int insertUser(User user);-->
    <insert id="insertUser">
        insert into t_user values(null, #{username}, #{password}, #{age}, #{sex}, #{email})
    </insert>

    <select id="checkLoginByParam" resultType="User">
        select * from t_user where username=#{username} and password=#{password}
    </select>
</mapper>
```

```java
import java.io.IOException;
import java.io.InputStream;

public class SqlSessionUtils {
    public static SqlSession getSqlSession() {
        SqlSession sqlSession = null;
        try {
            InputStream is = Resources.getResourceAsStream("mybatis-config.xml");
            SqlSessionFactory sqlSessionFactory = new SqlSessionFactoryBuilder().build(is);
            sqlSession = sqlSessionFactory.openSession(true);

        } catch (IOException e) {
            throw new RuntimeException(e);
        }
        return sqlSession;
    }
}
```

```java
public interface ParameterMapper {

    List<User> getAllUser();

    User getUserByName(String name);

    User checkLogin(String username, String password);

    User checkLoginByMap(Map map);

    int insertUser(User user);

    User checkLoginByParam(@Param("username") String username, @Param("password") String password);

}
```

```java
public class ParameterMapperTest {

    @Test
    public void testGetAllUsers() {
        SqlSession sqlSession = SqlSessionUtils.getSqlSession();
        ParameterMapper mapper = sqlSession.getMapper(ParameterMapper.class);

        List<User> allUsers = mapper.getAllUser();
        allUsers.forEach(user -> System.out.println(user));
    }

    // JDBC方法
    @Test
    public void testJDBC() throws ClassNotFoundException, SQLException {
        String userName = "wangwu";
        Class.forName("");
        Connection connection = DriverManager.getConnection("", "", "");
        PreparedStatement preparedStatement = connection.prepareStatement("select * from t_user where username=?");
        preparedStatement.setString(1, userName);
    }

    /**
     * MyBatis获取参数的两种方式：${} 和 #{}
     * ${} 本质字符串拼接
     * #{} 本质占位符赋值,防止注入攻击
     */

    // 1. 当mapper接口方法的参数为单个字面量类型 可以通过${} #{}以任意名称获取参数值，但${}要注意单引号
    @Test
    public void testGetUserByName() {
        SqlSession sqlSession = SqlSessionUtils.getSqlSession();
        ParameterMapper mapper = sqlSession.getMapper(ParameterMapper.class);
        String userName = "wangwu";
        User user = mapper.getUserByName(userName);
        System.out.println(user);
    }

    /**
     * 2. 当mapper接口方法参数为多个时，会将参数放在一个map集合中以两种方式储存
     *      a. arg0,arg1...为键，参数为值
     *      b. param1, param2...为键，参数为值
    */
    @Test
    public void testCheckLogin() {
        SqlSession sqlSession = SqlSessionUtils.getSqlSession();
        ParameterMapper mapper = sqlSession.getMapper(ParameterMapper.class);
        String userName = "wangwu";
        String password = "123";
        User user = mapper.checkLogin(userName, password);
        System.out.println(user);
    }

    // 3.若mapper接口方法的参数有多个时，可以手动将这些参数放到一个map中存储
    @Test
    public void testCheckLoginByMap() {
        SqlSession sqlSession = SqlSessionUtils.getSqlSession();
        ParameterMapper mapper = sqlSession.getMapper(ParameterMapper.class);
        HashMap<String, String> map = new HashMap<>();
        map.put("username","wangwu");
        map.put("password","123");
        User user = mapper.checkLoginByMap(map);
        System.out.println(user);
    }

    /**
     * 4. mapper接口参数为实体类类型
     * 需要通过#{} ${} 以属性的方式访问属性值
     */
    @Test
    public void testInsertUser() {
        SqlSession sqlSession = SqlSessionUtils.getSqlSession();
        ParameterMapper mapper = sqlSession.getMapper(ParameterMapper.class);
        int i = mapper.insertUser(new User(null, "lisi", "123", 23, "b", "213@qwe.com"));
        System.out.println(i);
    }

    /**
     * 5. 使用@param注解, 会将参数放在一个map集合中以两种方式储存
     *      a. @Param注解的值为键,参数为值
     *      b. param1, param2...为键，参数为值
     */
    @Test
    public void testCheckLoginByParam() {
        SqlSession sqlSession = SqlSessionUtils.getSqlSession();
        ParameterMapper mapper = sqlSession.getMapper(ParameterMapper.class);
        User i = mapper.checkLoginByParam("wangwu", "123");
        System.out.println(i);
    }
}
```
